        <script>
            function drabina_ulanicki(d, x) {

                let v =  Math.sqrt((d*d)+(x*x));

                document.getElementsByName("drabina")[0].value = v;
                document.getElementsByName("wyprowadz_drabina_ula")[0].innerHTML = `Długość Drabiny = ${v.toFixed(2)}`;
                alert(`Długość Drabiny = ${v.toFixed(2)}`);
            }

            function alfa_ulanicki(d, x) {

                let sinA = d / Math.sqrt((d*d)+(x*x));
                let aRad = Math.asin(sinA);
                let aDeg = aRad * (180 / Math.PI);

                document.getElementsByName("alfa")[0].value = aDeg.toFixed(2);
                document.getElementsByName("wyprowadz_alfa_ula")[0].innerHTML = `Alfa = ${aDeg.toFixed(2)}`;
                alert(`Alfa = ${aDeg.toFixed(2)}`)
            }
        </script>
