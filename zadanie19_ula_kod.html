<!DOCTYPE html>
<html>
    <head>
        <title>Wszystko o kwadracie</title>
        <meta charset="utf-8">
        <script type="text/javascript" defer>
                
            function pole_ula() {
                let a_ula = parseFloat(document.getElementById("dlugosc").value);
                let b_ula = a_ula*a_ula;
                document.getElementById("nazwa").innerHTML = `Pole kwadratu=${b_ula.toFixed(2)} cm^2`;
            }

            function obwod_ula() {
                let a_ula = parseFloat(document.getElementById("dlugosc").value);
                let b_ula = 4*a_ula;
                document.getElementById("nazwa").innerHTML = `Obwód kwadratu=${b_ula.toFixed(2)} cm`;
            }

        </script>
    </head>
    <body>
        <h1>Wszystko o KWADRACIE</h1><br><br>
        Podaj długość boku:
        <input type="text" id="dlugosc" value="podaj dlugosc">cm<br><br>

        <button name="pole" onclick="pole_ula()">Oblicz pole</button><br><br>
        <button name="obwod" onclick="obwod_ula()">Oblicz obwód</button><br><br>
        
        <div style="color: red; font-size: 50px;" id="nazwa"></div>
    </body>
